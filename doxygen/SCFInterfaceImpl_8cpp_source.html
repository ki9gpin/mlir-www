22 struct ExecuteRegionOpInterface
23  : public BufferizableOpInterface::ExternalModel<ExecuteRegionOpInterface,
24  scf::ExecuteRegionOp> {
25  SmallVector<OpOperand *>
26  getAliasingOpOperand(Operation *op, OpResult opResult,
27  BufferizationState &state) const {
28  // ExecuteRegionOps do not have tensor OpOperands. The yielded value can be
29  // any SSA value that is in scope. To allow for use-def chain traversal
30  // through ExecuteRegionOps in the analysis, the corresponding yield value
31  // is considered to be aliasing with the result.
32  auto executeRegionOp = cast<scf::ExecuteRegionOp>(op);
33  size_t resultNum = std::distance(op->getOpResults().begin(),
34  llvm::find(op->getOpResults(), opResult));
35  assert(executeRegionOp.getRegion().getBlocks().size() == 1 &&
36  "expected exactly 1 block");
37  auto yieldOp = dyn_cast<scf::YieldOp>(
38  executeRegionOp.getRegion().front().getTerminator());
39  assert(yieldOp && "expected scf.yield terminator in scf.execute_region");
40  return {&yieldOp->getOpOperand(resultNum)};
41  }

44  BufferizationState &state) const {
45  // ExecuteRegionOp results always bufferize in-place. Since they have no
46  // OpOperands, they are mostly ignored by the analysis once alias sets are
47  // set up.

51  // TODO: For better bufferization results, this could return `true` only if
52  // there is a memory write in the region.

84 struct IfOpInterface
85  : public BufferizableOpInterface::ExternalModel<IfOpInterface, scf::IfOp> {
86  SmallVector<OpOperand *>
87  getAliasingOpOperand(Operation *op, OpResult opResult,
88  BufferizationState &state) const {
89  // IfOps do not have tensor OpOperands. The yielded value can be any SSA
90  // value that is in scope. To allow for use-def chain traversal through
91  // IfOps in the analysis, both corresponding yield values from the then/else
92  // branches are considered to be aliasing with the result.
93  auto ifOp = cast<scf::IfOp>(op);
94  size_t resultNum = std::distance(op->getOpResults().begin(),
95  llvm::find(op->getOpResults(), opResult));
96  return {&ifOp.thenYield()->getOpOperand(resultNum),
97  &ifOp.elseYield()->getOpOperand(resultNum)};
98  }
99 
100  // TODO: For better bufferization results, this could return `true` only if
101  // there is a memory write in one (or both) of the branches. Since this is not
102  // allowed at the moment, we should never encounter scf.ifs that yield
103  // unmodified tensors. Such scf.yield ops could just fold away.
104  bool isMemoryWrite(Operation *op, OpResult opResult,
105  BufferizationState &state) const {
106  // IfOp results are always considered memory writes in the analysis. This
107  // design decision simplifies the analysis considerably. E.g., consider the
108  // following test case:
109  //
110  // %0 = "some_writing_op" : tensor<?xf32>
111  // %r = scf.if %c -> (tensor<?xf32>) {
112  // scf.yield %0
113  // } else {
114  // %1 = "another_writing_op"(%0) : tensor<?xf32>
115  // }
116  // "some_reading_op"(%r)
117  //
118  // "another_writing_op" in the above example should be able to bufferize
119  // inplace in the absence of another read of %0. However, if the scf.if op
120  // would not be considered a "write", the analysis would detect the
121  // following conflict:
122  //
123  // * read = some_reading_op
124  // * lastWrite = %0 (Note: The last write of %r would be a set: {%0, %1}.)
125  // * conflictingWrite = %1
126  //
127  // For more details, check the "scf.IfOp" section of the design document.
128  return true;
129  }

138  LogicalResult bufferize(Operation *op, OpBuilder &b,
139  BufferizationState &state) const {
140  auto ifOp = cast<scf::IfOp>(op);
141 
142  // Use IRRewriter instead of OpBuilder because it has additional helper
143  // functions.
144  IRRewriter rewriter(op->getContext());
145  rewriter.setInsertionPoint(ifOp);
146 
147  // Compute new types of the bufferized scf.if op.
148  SmallVector<Type> newTypes;
149  for (Type returnType : ifOp->getResultTypes()) {
150  if (returnType.isa<TensorType>()) {
151  assert(returnType.isa<RankedTensorType>() &&
152  "unsupported unranked tensor");
153  newTypes.push_back(
154  getDynamicMemRefType(returnType.cast<RankedTensorType>()));
155  } else {
156  newTypes.push_back(returnType);
157  }
158  }
159 
160  // Create new op.
161  auto newIfOp =
162  rewriter.create<scf::IfOp>(ifOp.getLoc(), newTypes, ifOp.getCondition(),
163  /*withElseRegion=*/true);
164 
165  // Remove terminators.
166  if (!newIfOp.thenBlock()->empty()) {
167  rewriter.eraseOp(newIfOp.thenBlock()->getTerminator());
168  rewriter.eraseOp(newIfOp.elseBlock()->getTerminator());
169  }
170 
171  // Move over then/else blocks.
172  rewriter.mergeBlocks(ifOp.thenBlock(), newIfOp.thenBlock());
173  rewriter.mergeBlocks(ifOp.elseBlock(), newIfOp.elseBlock());
174 
175  // Update scf.yield of new then-block.
176  auto thenYieldOp = cast<scf::YieldOp>(newIfOp.thenBlock()->getTerminator());
177  rewriter.setInsertionPoint(thenYieldOp);
178  SmallVector<Value> thenYieldValues;
179  for (OpOperand &operand : thenYieldOp->getOpOperands()) {
180  if (operand.get().getType().isa<TensorType>()) {
181  Value toMemrefOp = rewriter.create<bufferization::ToMemrefOp>(
182  operand.get().getLoc(), newTypes[operand.getOperandNumber()],
183  operand.get());
184  operand.set(toMemrefOp);
185  }
186  }
187 
188  // Update scf.yield of new else-block.
189  auto elseYieldOp = cast<scf::YieldOp>(newIfOp.elseBlock()->getTerminator());
190  rewriter.setInsertionPoint(elseYieldOp);
191  SmallVector<Value> elseYieldValues;
192  for (OpOperand &operand : elseYieldOp->getOpOperands()) {
193  if (operand.get().getType().isa<TensorType>()) {
194  Value toMemrefOp = rewriter.create<bufferization::ToMemrefOp>(
195  operand.get().getLoc(), newTypes[operand.getOperandNumber()],
196  operand.get());
197  operand.set(toMemrefOp);
198  }
199  }
200 
201  // Replace op results.
202  state.replaceOp(op, newIfOp->getResults());
203 
204  // Bufferize then/else blocks.
205  if (failed(comprehensive_bufferize::bufferize(newIfOp.thenBlock(), state)))
206  return failure();
207  if (failed(comprehensive_bufferize::bufferize(newIfOp.elseBlock(), state)))
208  return failure();
209 
210  return success();
211  }

213  BufferRelation bufferRelation(Operation *op, OpResult opResult,
214  const BufferizationAliasInfo &aliasInfo,
215  BufferizationState &state) const {
216  // IfOp results are equivalent to their corresponding yield values if both
217  // yield values are equivalent to each other.
218  auto bufferizableOp = cast<BufferizableOpInterface>(op);
219  SmallVector<OpOperand *> yieldValues =
220  bufferizableOp.getAliasingOpOperand(opResult, state);
221  assert(yieldValues.size() == 2 && "expected 2 yield values");
222  bool equivalentYields = aliasInfo.areEquivalentBufferizedValues(
223  yieldValues[0]->get(), yieldValues[1]->get());
224  return equivalentYields ? BufferRelation::Equivalent : BufferRelation::None;
225  }

255  BufferRelation bufferRelation(Operation *op, OpResult opResult,
256  const BufferizationAliasInfo &aliasInfo,
257  BufferizationState &state) const {

259  // corresponding iter_args and yield values are equivalent.
260  auto forOp = cast<scf::ForOp>(op);
261  OpOperand &forOperand = forOp.getOpOperandForResult(opResult);

266  return equivalentYield ? BufferRelation::Equivalent : BufferRelation::None;

269  bool isWritable(Operation *op, Value value, BufferizationState &state) const {
270  // Interestingly, scf::ForOp's bbArg can **always** be viewed
271  // inplace from the perspective of ops nested under:
272  // 1. Either the matching iter operand is not bufferized inplace and an
273  // alloc + optional copy makes the bbArg itself inplaceable.
274  // 2. Or the matching iter operand is bufferized inplace and bbArg just
275  // bufferizes to that too.
276  return true;

279  LogicalResult bufferize(Operation *op, OpBuilder & /*b*/,
280  BufferizationState &state) const {
281  auto forOp = cast<scf::ForOp>(op);
282  Block *oldLoopBody = &forOp.getLoopBody().front();
283 
284  // Use IRRewriter instead of OpBuilder because it has additional helper
285  // functions.
286  IRRewriter rewriter(op->getContext());
287  rewriter.setInsertionPoint(forOp);
288 
289  // Indices of all iter_args that have tensor type. These are the ones that
290  // are bufferized.
291  DenseSet<int64_t> indices;
292  for (const auto &it : llvm::enumerate(forOp.getInitArgs()))
293  if (it.value().getType().isa<TensorType>())
294  indices.insert(it.index());
295 
296  // Given a range of values, apply `func` to those marked in `indices`.
297  // Otherwise, store the unmodified value in the result vector.
298  auto convert = [&](ValueRange values,
299  std::function<Value(Value, int64_t)> func) {
300  SmallVector<Value> result;
301  for (const auto &it : llvm::enumerate(values)) {
302  size_t idx = it.index();
303  Value val = it.value();
304  result.push_back(indices.contains(idx) ? func(val, idx) : val);
305  }
306  return result;
307  };
308 
309  // Construct a new scf.for op with memref instead of tensor values.
310  SmallVector<Value> initArgs =
311  convert(forOp.getInitArgs(), [&](Value val, int64_t index) {
312  return state.getResultBuffer(forOp->getOpResult(index));
313  });
314  auto newForOp = rewriter.create<scf::ForOp>(
315  forOp.getLoc(), forOp.getLowerBound(), forOp.getUpperBound(),
316  forOp.getStep(), initArgs);
317  Block *loopBody = &newForOp.getLoopBody().front();
318 
319  // Set up new iter_args. The loop body uses tensors, so wrap the (memref)
320  // iter_args of the new loop in ToTensorOps.
321  rewriter.setInsertionPointToStart(loopBody);
322  SmallVector<Value> iterArgs =
323  convert(newForOp.getRegionIterArgs(), [&](Value val, int64_t index) {
324  return rewriter.create<bufferization::ToTensorOp>(val.getLoc(), val);
325  });
326  iterArgs.insert(iterArgs.begin(), newForOp.getInductionVar());
327 
328  // Erase terminator if present.
329  if (iterArgs.size() == 1)
330  rewriter.eraseOp(loopBody->getTerminator());
331 
332  // Move loop body to new loop.
333  rewriter.mergeBlocks(oldLoopBody, loopBody, iterArgs);
334 
335  // Update scf.yield of new loop.
336  auto yieldOp = cast<scf::YieldOp>(loopBody->getTerminator());
337  rewriter.setInsertionPoint(yieldOp);
338  SmallVector<Value> yieldValues =
339  convert(yieldOp.getResults(), [&](Value val, int64_t index) {
340  return rewriter.create<bufferization::ToMemrefOp>(
341  val.getLoc(), initArgs[index].getType(), val);
342  });
343  yieldOp.getResultsMutable().assign(yieldValues);
344 
345  // Replace loop results.
346  state.replaceOp(op, newForOp->getResults());
347 
348  // Bufferize loop body.
349  if (failed(comprehensive_bufferize::bufferize(loopBody, state)))
350  return failure();
351 
352  return success();
353  }
354 };

359 LogicalResult mlir::linalg::comprehensive_bufferize::scf_ext::
360  AssertDestinationPassingStyle::run(Operation *op, BufferizationState &state,
361  BufferizationAliasInfo &aliasInfo,
362  SmallVector<Operation *> &newOps) {
363  LogicalResult status = success();
364  op->walk([&](scf::YieldOp yieldOp) {
365  if (auto forOp = dyn_cast<scf::ForOp>(yieldOp->getParentOp())) {
366  for (OpOperand &operand : yieldOp->getOpOperands()) {
367  auto tensorType = operand.get().getType().dyn_cast<TensorType>();
368  if (!tensorType)
369  continue;
370 
371  OpOperand &forOperand = forOp.getOpOperandForResult(
372  forOp->getResult(operand.getOperandNumber()));
373  auto bbArg = forOp.getRegionIterArgForOpOperand(forOperand);
374  if (!aliasInfo.areEquivalentBufferizedValues(operand.get(), bbArg)) {
375  // TODO: this could get resolved with copies but it can also turn into
376  // swaps so we need to be careful about order of copies.
377  status =
378  yieldOp->emitError()
379  << "Yield operand #" << operand.getOperandNumber()
380  << " does not bufferize to an equivalent buffer to the matching"
381  << " enclosing scf::for operand";
382  return WalkResult::interrupt();
383  }
384  }
385  }
386 
387  if (auto ifOp = dyn_cast<scf::IfOp>(yieldOp->getParentOp())) {
388  // IfOps are in destination passing style if all yielded tensors are
389  // a value or equivalent to a value that is defined outside of the IfOp.
390  for (OpOperand &operand : yieldOp->getOpOperands()) {
391  auto tensorType = operand.get().getType().dyn_cast<TensorType>();
392  if (!tensorType)
393  continue;
394 
395  bool foundOutsideEquivalent = false;
396  aliasInfo.applyOnEquivalenceClass(operand.get(), [&](Value value) {
397  Operation *valueOp = value.getDefiningOp();
398  if (value.isa<BlockArgument>())
399  valueOp = value.cast<BlockArgument>().getOwner()->getParentOp();
400 
401  bool inThenBlock = ifOp.thenBlock()->findAncestorOpInBlock(*valueOp);
402  bool inElseBlock = ifOp.elseBlock()->findAncestorOpInBlock(*valueOp);
403 
404  if (!inThenBlock && !inElseBlock)
405  foundOutsideEquivalent = true;
406  });
407 
408  if (!foundOutsideEquivalent) {
409  status = yieldOp->emitError()
410  << "Yield operand #" << operand.getOperandNumber()
411  << " does not bufferize to a buffer that is equivalent to a"
412  << " buffer defined outside of the scf::if op";
413  return WalkResult::interrupt();
414  }
415  }
416  }
417 
418  return WalkResult::advance();
419  });
420  return status;
421 }
