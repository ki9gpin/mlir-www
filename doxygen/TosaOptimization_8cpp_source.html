37 struct Conv2DIsFullyConnected : public OpRewritePattern<tosa::Conv2DOp> {
38  explicit Conv2DIsFullyConnected(MLIRContext *context)
39  : OpRewritePattern(context) {}
40 
41  LogicalResult matchAndRewrite(tosa::Conv2DOp op,
42  PatternRewriter &rewriter) const override {
43  Value input = op.input();
44  Value weight = op.weight();
45  ShapedType inputType = input.getType().cast<ShapedType>();
46  ShapedType weightType = weight.getType().cast<ShapedType>();
47  ShapedType resultType = op.getType().cast<ShapedType>();
48 
49  if (!inputType.hasStaticShape() || !weightType.hasRank()) {
50  return failure();
51  }
52 
53  // Stride must be 1 for this optimization.
54  for (Attribute stride : op.stride().getValue()) {
55  if (!stride.cast<IntegerAttr>().getValue().isOne()) {
56  return failure();
57  }
58  }
59 
60  // Only works for a 1x1 kernel.
61  ArrayRef<int64_t> weightShape = weightType.getShape();
62  if (weightShape[1] != 1 || weightShape[2] != 1) {
63  return failure();
64  }
65 
66  // Reshape input to [N,IH,IW,IC] -> [N * IH * IW, IC].
67  ArrayRef<int64_t> inputShape = inputType.getShape();
68  llvm::SmallVector<int64_t, 2> revisedInputShape{
69  inputShape[0] * inputShape[1] * inputShape[2], inputShape[3]};
70  auto revisedInputShapeType = RankedTensorType::get(
71  revisedInputShape,
72  input.getType().dyn_cast<RankedTensorType>().getElementType());
73  auto reshapedInput = rewriter
74  .create<tosa::ReshapeOp>(
75  op.getLoc(), revisedInputShapeType, input,
76  rewriter.getI64ArrayAttr(revisedInputShape))
77  .getResult();
78 
79  // Reshape kernel to [OC,KH,KW,IC] -> [OC, IC].
80  llvm::SmallVector<int64_t, 2> revisedWeightShape{weightShape[0],
81  weightShape[3]};
82  auto revisedWeightShapeType = RankedTensorType::get(
83  revisedWeightShape,
84  weight.getType().dyn_cast<RankedTensorType>().getElementType());
85  auto reshapedWeight = rewriter
86  .create<tosa::ReshapeOp>(
87  op.getLoc(), revisedWeightShapeType, weight,
88  rewriter.getI64ArrayAttr(revisedWeightShape))
89  .getResult();
90 
91  // Perform a fully connected network over the reshaped input and weight.
92  llvm::SmallVector<int64_t, 2> fullyConnectedShape{
93  inputShape[0] * inputShape[1] * inputShape[2], weightShape[0]};
94  auto fullyConnectedShapeType = RankedTensorType::get(
95  fullyConnectedShape,
96  resultType.dyn_cast<ShapedType>().getElementType());
97 
98  Value fullyConnectedValue;
99  if (op.quantization_info()) {
100  fullyConnectedValue =
101  rewriter
102  .create<tosa::FullyConnectedOp>(
103  op.getLoc(), fullyConnectedShapeType, reshapedInput,
104  reshapedWeight, op.bias(), op.quantization_info().getValue())
105  .getResult();
106  } else {
107  fullyConnectedValue = rewriter
108  .create<tosa::FullyConnectedOp>(
109  op.getLoc(), fullyConnectedShapeType,
110  reshapedInput, reshapedWeight, op.bias())
111  .getResult();
112  }
113 
114  // Reshape output to [N, IH, IW, OC].
115  llvm::SmallVector<int64_t, 4> outputShape{inputShape[0], inputShape[1],
116  inputShape[2], weightShape[0]};
117  rewriter.replaceOpWithNewOp<tosa::ReshapeOp>(
118  op, resultType, fullyConnectedValue,
119  rewriter.getI64ArrayAttr(outputShape));
120  return success();
121  }
122 };

124 struct DepthwiseConv2DIsMul : public OpRewritePattern<tosa::DepthwiseConv2DOp> {
125  explicit DepthwiseConv2DIsMul(MLIRContext *context)
126  : OpRewritePattern(context) {}
127 
128  LogicalResult matchAndRewrite(tosa::DepthwiseConv2DOp op,
129  PatternRewriter &rewriter) const override {
130  Value input = op.input();
131  Value weight = op.weight();
132  ShapedType inputType = input.getType().cast<ShapedType>();
133  ShapedType weightType = weight.getType().cast<ShapedType>();
134  ShapedType resultType = op.output().getType().cast<ShapedType>();
135  Type inputEType = inputType.getElementType();
136 
137  if (!(inputType.hasStaticShape() && weightType.hasStaticShape() &&
138  resultType.hasStaticShape())) {
139  return failure();
140  }
141 
142  // Quantization information needs to still be performed.
143  if (op.quantization_info() || !inputEType.isa<FloatType>()) {
144  return failure();
145  }
146 
147  // Stride must be 1 for this optimization.
148  for (Attribute stride : op.stride().getValue()) {
149  if (!stride.cast<IntegerAttr>().getValue().isOne()) {
150  return failure();
151  }
152  }
153 
154  // Only works for a 1x1 kernel.
155  ArrayRef<int64_t> weightShape = weightType.getShape();
156  if (weightShape[0] != 1 || weightShape[1] != 1) {
157  return failure();
158  }
159 
160  // Reshape input to [N, H, W, C] -> [N, H, W, C, 1].
161  ArrayRef<int64_t> inputShape = inputType.getShape();
162  llvm::SmallVector<int64_t, 2> revisedInputShape{
163  inputShape[0], inputShape[1], inputShape[2], inputShape[3], 1};
164  auto revisedInputShapeType = RankedTensorType::get(
165  revisedInputShape,
166  input.getType().dyn_cast<RankedTensorType>().getElementType());
167  auto reshapedInput = rewriter
168  .create<tosa::ReshapeOp>(
169  op.getLoc(), revisedInputShapeType, input,
170  rewriter.getI64ArrayAttr(revisedInputShape))
171  .getResult();
172 
173  // Reshape kernel to [KH, KW, C, M] -> [1, 1, 1, C, M].
174  llvm::SmallVector<int64_t, 2> revisedWeightShape{1, 1, 1, weightShape[2],
175  weightShape[3]};
176  auto revisedWeightShapeType = RankedTensorType::get(
177  revisedWeightShape,
178  weight.getType().dyn_cast<RankedTensorType>().getElementType());
179  auto reshapedWeight = rewriter
180  .create<tosa::ReshapeOp>(
181  op.getLoc(), revisedWeightShapeType, weight,
182  rewriter.getI64ArrayAttr(revisedWeightShape))
183  .getResult();
184 
185  // Perform an elementwise mul over the reshaped input and weight.
186  llvm::SmallVector<int64_t, 2> mulShape{inputShape[0], inputShape[1],
187  inputShape[2], inputShape[3],
188  weightShape[3]};
189  auto mulShapeType = RankedTensorType::get(
190  mulShape,
191  weight.getType().dyn_cast<RankedTensorType>().getElementType());
192  Value mulValue =
193  rewriter
194  .create<tosa::MulOp>(op.getLoc(), mulShapeType, reshapedInput,
195  reshapedWeight, /*shift=*/0)
196  .getResult();
197 
198  // Reshape output to [N, H, W, C * M].
199  auto outputShape = op.output().getType().cast<ShapedType>().getShape();
200  auto outputShapeType = RankedTensorType::get(
201  outputShape,
202  input.getType().dyn_cast<RankedTensorType>().getElementType());
203  auto outputValue =
204  rewriter.create<tosa::ReshapeOp>(op.getLoc(), outputShapeType, mulValue,
205  rewriter.getI64ArrayAttr(outputShape));
206 
207  // Add in the bias.
208  rewriter
209  .replaceOpWithNewOp<tosa::AddOp>(op, outputShapeType, outputValue,
210  op.bias())
211  .getResult();
212  return success();
213  }
214 };

216 class TosaOptimization : public PassWrapper<TosaOptimization, FunctionPass> {
217 public:
218  explicit TosaOptimization() = default;
219  void runOnFunction() override;

221  StringRef getArgument() const final { return PASS_NAME; }

227 void TosaOptimization::runOnFunction() {
228  OwningRewritePatternList patterns(&getContext());
229 
230  patterns.insert<Conv2DIsFullyConnected>(&getContext());
231  patterns.insert<DepthwiseConv2DIsMul>(&getContext());
232 
233  auto func = getFunction();
234  if (applyPatternsAndFoldGreedily(func, std::move(patterns)).failed()) {
235  signalPassFailure();
236  }
237 }
