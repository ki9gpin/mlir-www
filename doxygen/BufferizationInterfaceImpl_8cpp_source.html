28 /// ToMemrefOp casts a tensor into a memref. The resulting memref is the memory
29 /// location of the incoming tensor once it will be bufferized. In the anlysis,
30 /// the incoming tensor is assumed to bufferize to a memory read and to an
31 /// inplace memory write, since it is unknown what will happen to the resulting
32 /// memref.
33 ///
34 /// Note: ToMemrefOp / ToTensorOp are temporary ops that are inserted at the
35 /// bufferization boundary. When bufferization is complete, there should be no
36 /// such ops left over. If `allowUnknownOps`, such ops may be part of the
37 /// resulting IR, but such IR may no longer be bufferizable by Comprehensive
38 /// Bufferize.
39 struct ToMemrefOpInterface
40  : public BufferizableOpInterface::ExternalModel<ToMemrefOpInterface,
41  bufferization::ToMemrefOp> {
42  bool bufferizesToMemoryRead(Operation *op, OpOperand &opOperand,
43  BufferizationState &state) const {
44  // It is unknown whether the resulting MemRef will be read or not.
45  return true;
46  }
47 
48  OpResult getAliasingOpResult(Operation *op, OpOperand &opOperand,
49  BufferizationState &state) const {
50  return OpResult();
51  }
52 
53  LogicalResult bufferize(Operation *op, OpBuilder &b,
54  BufferizationState &state) const {
55  auto toMemrefOp = cast<bufferization::ToMemrefOp>(op);
56 
57  // Fold to_memref(to_tensor(x)) to x.
58  if (auto toTensorOp =
59  toMemrefOp.tensor().getDefiningOp<bufferization::ToTensorOp>()) {
60  toMemrefOp.replaceAllUsesWith(toTensorOp.memref());
61  toMemrefOp.erase();
62  return success();
63  }
64 
65  // If a ToMemrefOp's tensor operand has not been bufferized yet, the op
66  // remains unchanged. All IR up to this ToMemrefOp has already been
67  // bufferized, unless there were unknown ops that could be bufferized.
68  assert((isFunctionArgument(toMemrefOp.tensor()) ||
69  state.getOptions().allowUnknownOps) &&
70  "expected that tensor is mapped");
71 
72  return success();
73  }
74 };

76 /// ToTensorOp conceptually loads a tensor from a memory location. Such ops do
77 /// not lower any further, and they should have disappeared by the time the
78 /// input is fully bufferized.
79 ///
80 /// The analysis has no information about the memref that is loaded from by the
81 /// ToTensorOp. We have to assume that the loaded tensor may after bufferization
82 /// potentially alias with any other bufferized tensor. Since ToTensorOp and
83 /// ToMemrefOp have no aliasing OpOperand/OpResult pairs, this cannot be encoded
84 /// directly in the analysis. However, declaring ToTensorOp results as not
85 /// writable also enforces a buffer copy and has the same effect.
86 struct ToTensorOpInterface
87  : public BufferizableOpInterface::ExternalModel<ToTensorOpInterface,
88  bufferization::ToTensorOp> {
89  LogicalResult bufferize(Operation *op, OpBuilder &b,
90  BufferizationState &state) const {
91  return success();
92  }
93 
94  bool isWritable(Operation *op, Value value, BufferizationState &state) const {
95  // It is unknown whether the MemRef operand is writable or not.
96  return false;
97  }
98 };
