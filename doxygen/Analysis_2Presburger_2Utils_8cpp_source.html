19 /// Check if the pos^th identifier can be represented as a division using upper
20 /// bound inequality at position `ubIneq` and lower bound inequality at position
21 /// `lbIneq`.
22 ///
23 /// Let `id` be the pos^th identifier, then `id` is equivalent to
24 /// `expr floordiv divisor` if there are constraints of the form:
25 /// 0 <= expr - divisor * id <= divisor - 1
26 /// Rearranging, we have:
27 /// divisor * id - expr + (divisor - 1) >= 0 <-- Lower bound for 'id'
28 /// -divisor * id + expr >= 0 <-- Upper bound for 'id'
29 ///
30 /// For example:
31 /// 32*k >= 16*i + j - 31 <-- Lower bound for 'k'
32 /// 32*k <= 16*i + j <-- Upper bound for 'k'
33 /// expr = 16*i + j, divisor = 32
34 /// k = ( 16*i + j ) floordiv 32
35 ///
36 /// 4q >= i + j - 2 <-- Lower bound for 'q'
37 /// 4q <= i + j + 1 <-- Upper bound for 'q'
38 /// expr = i + j + 1, divisor = 4
39 /// q = (i + j + 1) floordiv 4
40 //
41 /// This function also supports detecting divisions from bounds that are
42 /// strictly tighter than the division bounds described above, since tighter
43 /// bounds imply the division bounds. For example:
44 /// 4q - i - j + 2 >= 0 <-- Lower bound for 'q'
45 /// -4q + i + j >= 0 <-- Tight upper bound for 'q'
46 ///
47 /// To extract floor divisions with tighter bounds, we assume that that the
48 /// constraints are of the form:
49 /// c <= expr - divisior * id <= divisor - 1, where 0 <= c <= divisor - 1
50 /// Rearranging, we have:
51 /// divisor * id - expr + (divisor - 1) >= 0 <-- Lower bound for 'id'
52 /// -divisor * id + expr - c >= 0 <-- Upper bound for 'id'
53 ///
54 /// If successful, `expr` is set to dividend of the division and `divisor` is
55 /// set to the denominator of the division.
56 static LogicalResult getDivRepr(const IntegerPolyhedron &cst, unsigned pos,
57  unsigned ubIneq, unsigned lbIneq,
58  SmallVector<int64_t, 8> &expr,
59  unsigned &divisor) {
60 
61  assert(pos <= cst.getNumIds() && "Invalid identifier position");
62  assert(ubIneq <= cst.getNumInequalities() &&
63  "Invalid upper bound inequality position");
64  assert(lbIneq <= cst.getNumInequalities() &&
65  "Invalid upper bound inequality position");
66 
67  // Extract divisor from the lower bound.
68  divisor = cst.atIneq(lbIneq, pos);
69 
70  // First, check if the constraints are opposite of each other except the
71  // constant term.
72  unsigned i = 0, e = 0;
73  for (i = 0, e = cst.getNumIds(); i < e; ++i)
74  if (cst.atIneq(ubIneq, i) != -cst.atIneq(lbIneq, i))
75  break;
76 
77  if (i < e)
78  return failure();
79 
80  // Then, check if the constant term is of the proper form.
81  // Due to the form of the upper/lower bound inequalities, the sum of their
82  // constants is `divisor - 1 - c`. From this, we can extract c:
83  int64_t constantSum = cst.atIneq(lbIneq, cst.getNumCols() - 1) +
84  cst.atIneq(ubIneq, cst.getNumCols() - 1);
85  int64_t c = divisor - 1 - constantSum;
86 
87  // Check if `c` satisfies the condition `0 <= c <= divisor - 1`. This also
88  // implictly checks that `divisor` is positive.
89  if (!(c >= 0 && c <= divisor - 1))
90  return failure();
91 
92  // The inequality pair can be used to extract the division.
93  // Set `expr` to the dividend of the division except the constant term, which
94  // is set below.
95  expr.resize(cst.getNumCols(), 0);
96  for (i = 0, e = cst.getNumIds(); i < e; ++i)
97  if (i != pos)
98  expr[i] = cst.atIneq(ubIneq, i);
99 
100  // From the upper bound inequality's form, its constant term is equal to the
101  // constant term of `expr`, minus `c`. From this,
102  // constant term of `expr` = constant term of upper bound + `c`.
103  expr.back() = cst.atIneq(ubIneq, cst.getNumCols() - 1) + c;
104 
105  return success();
106 }

108 /// Check if the pos^th identifier can be expressed as a floordiv of an affine
109 /// function of other identifiers (where the divisor is a positive constant).
110 /// `foundRepr` contains a boolean for each identifier indicating if the
111 /// explicit representation for that identifier has already been computed.
112 /// Returns the upper and lower bound inequalities using which the floordiv can
113 /// be computed. If the representation could be computed, `dividend` and
114 /// `denominator` are set. If the representation could not be computed,
115 /// `llvm::None` is returned.
116 Optional<std::pair<unsigned, unsigned>> presburger_utils::computeSingleVarRepr(
117  const IntegerPolyhedron &cst, ArrayRef<bool> foundRepr, unsigned pos,
118  SmallVector<int64_t, 8> &dividend, unsigned &divisor) {
119  assert(pos < cst.getNumIds() && "invalid position");
120  assert(foundRepr.size() == cst.getNumIds() &&
121  "Size of foundRepr does not match total number of variables");
122 
123  SmallVector<unsigned, 4> lbIndices, ubIndices;
124  cst.getLowerAndUpperBoundIndices(pos, &lbIndices, &ubIndices);
125 
126  for (unsigned ubPos : ubIndices) {
127  for (unsigned lbPos : lbIndices) {
128  // Attempt to get divison representation from ubPos, lbPos.
129  if (failed(getDivRepr(cst, pos, ubPos, lbPos, dividend, divisor)))
130  continue;
131 
132  // Check if the inequalities depend on a variable for which
133  // an explicit representation has not been found yet.
134  // Exit to avoid circular dependencies between divisions.
135  unsigned c, f;
136  for (c = 0, f = cst.getNumIds(); c < f; ++c) {
137  if (c == pos)
138  continue;
139  if (!foundRepr[c] && dividend[c] != 0)
140  break;
141  }
142 
143  // Expression can't be constructed as it depends on a yet unknown
144  // identifier.
145  // TODO: Visit/compute the identifiers in an order so that this doesn't
146  // happen. More complex but much more efficient.
147  if (c < f)
148  continue;
149 
150  return std::make_pair(ubPos, lbPos);
151  }
152  }
153 
154  return llvm::None;
155 }
