259 /// Return true if the given value is read by an op that bufferizes to a memory
260 /// read. Also takes into account ops that create an alias but do not read by
261 /// themselves (e.g., ExtractSliceOp).
262 bool mlir::linalg::comprehensive_bufferize::BufferizationState::isValueRead(
263  Value value) {
264  SmallVector<OpOperand *> workingSet;
265  for (OpOperand &use : value.getUses())
266  workingSet.push_back(&use);
267 
268  while (!workingSet.empty()) {
269  OpOperand *uMaybeReading = workingSet.pop_back_val();
270  // Skip over all ops that neither read nor write (but create an alias).
271  if (bufferizesToAliasOnly(*uMaybeReading))
272  for (OpOperand &use : getAliasingOpResult(*uMaybeReading).getUses())
273  workingSet.push_back(&use);
274  if (bufferizesToMemoryRead(*uMaybeReading))
275  return true;
276  }
277 
278  return false;
279 }

281 // Starting from `value`, follow the use-def chain in reverse, always selecting
282 // the aliasing OpOperands. Find and return Values for which `condition`
283 // evaluates to true. OpOperands of such matching Values are not traversed any
284 // further.
285 llvm::SetVector<Value>
286 mlir::linalg::comprehensive_bufferize::BufferizationState::
287  findValueInReverseUseDefChain(Value value,
288  std::function<bool(Value)> condition) {
289  llvm::SetVector<Value> result, workingSet;
290  workingSet.insert(value);
291 
292  while (!workingSet.empty()) {
293  Value value = workingSet.pop_back_val();
294  if (condition(value) || value.isa<BlockArgument>()) {
295  result.insert(value);
296  continue;
297  }
298 
299  OpResult opResult = value.cast<OpResult>();
300  SmallVector<OpOperand *> opOperands = getAliasingOpOperand(opResult);
301  if (opOperands.empty() || !options.isOpAllowed(value.getDefiningOp())) {
302  result.insert(value);
303  continue;
304  }
305 
306  for (OpOperand *o : opOperands)
307  workingSet.insert(o->get());
308  }
309 
310  return result;
311 }

314 Value mlir::linalg::comprehensive_bufferize::BufferizationState::
315  findLastPrecedingWrite(Value value) {
316  SetVector<Value> result =
317  findValueInReverseUseDefChain(value, [&](Value value) {
318  Operation *op = value.getDefiningOp();
319  if (!op)
320  return true;

322  if (!bufferizableOp)
323  return true;
324  return bufferizableOp.isMemoryWrite(value.cast<OpResult>(), *this);
325  });

335 mlir::linalg::comprehensive_bufferize::BufferizationState::BufferizationState(
336  Operation *op, const BufferizationOptions &options)
337  : aliasInfo(op), options(options), builder(op->getContext()) {
338  // Set up alias sets for OpResults that must bufferize in-place. This should
339  // be done before making any other bufferization decisions.
340  op->walk([&](BufferizableOpInterface bufferizableOp) {
341  if (!options.isOpAllowed(bufferizableOp))
342  return WalkResult::skip();
343  for (OpResult opResult : bufferizableOp->getOpResults()) {
344  if (opResult.getType().isa<TensorType>())
345  if (bufferizableOp.mustBufferizeInPlace(opResult, *this)) {
346  SmallVector<OpOperand *> operands =
347  bufferizableOp.getAliasingOpOperand(opResult, *this);
348  assert(!operands.empty() &&
349  "expected that OpResult has aliasing OpOperand");
350  for (OpOperand *operand : operands)
351  aliasInfo.unionAliasSets(operand->get(), opResult);
352  aliasInfo.markInPlace(opResult);
353  }
354  }
355  return WalkResult::advance();
356  });
357 }

359 /// Return the result buffer (memref) for a given OpResult (tensor). Allocate
360 /// a new buffer and copy over data from the existing buffer if out-of-place
361 /// bufferization is necessary.
362 Value mlir::linalg::comprehensive_bufferize::BufferizationState::
363  getResultBuffer(OpResult result) {
364  OpBuilder::InsertionGuard guard(builder);
365  Operation *op = result.getOwner();
366  SmallVector<OpOperand *> aliasingOperands = getAliasingOpOperand(result);
367  assert(!aliasingOperands.empty() && "could not get aliasing OpOperand");
368  OpOperand *opOperand = aliasingOperands.front();
369  Value operand = opOperand->get();
370  Value operandBuffer = lookupBuffer(operand);
371  // Make sure that all OpOperands are the same buffer. If this is not the case,
372  // we would have to materialize a memref value.
373  // TODO: Should be looking for checking for "equivalent buffers" instead of
374  // operator== here, but equivalent buffers for scf.if yield values are not
375  // set up yet.
376  if (aliasingOperands.size() > 1 &&
377  !llvm::all_of(aliasingOperands, [&](OpOperand *o) {
378  return lookupBuffer(o->get()) == operandBuffer;
379  })) {
380  op->emitError("result buffer is ambiguous");
381  return Value();
382  }
383 
384  // If bufferizing out-of-place, allocate a new buffer.
385  if (!aliasInfo.isInPlace(result)) {
386  // Ops with multiple aliasing operands can currently not bufferize
387  // out-of-place.
388  assert(
389  aliasingOperands.size() == 1 &&
390  "ops with multiple aliasing OpOperands cannot bufferize out-of-place");
391  Location loc = op->getLoc();
392  // Move insertion point right after `operandBuffer`. That is where the
393  // allocation should be inserted (in the absence of allocation hoisting).
394  setInsertionPointAfter(builder, operandBuffer);
395  // Allocate the result buffer.
396  Value resultBuffer = createAllocDeallocPair(builder, loc, operandBuffer);
397  bool skipCopy = false;
398  // Do not copy if the last preceding write of `operand` is an op that does
399  // not write (skipping ops that merely create aliases). E.g., InitTensorOp.
400  // Note: If `findLastPrecedingWrite` reaches the end of the reverse SSA
401  // use-def chain, it returns that value, regardless of whether it is a
402  // memory write or not.
403  Value lastWrite = findLastPrecedingWrite(operand);
404  if (auto bufferizableOp = options.dynCastBufferizableOp(lastWrite))
405  if (!bufferizableOp.isMemoryWrite(lastWrite.cast<OpResult>(), *this))
406  skipCopy = true;
407  // Do not copy if the copied data is never read.
408  if (!isValueRead(result))
409  skipCopy = true;
410  // Do not copy if this op does not read the data, but writes it.
411  if (bufferizesToMemoryWrite(*opOperand) &&
412  !bufferizesToMemoryRead(*opOperand))
413  skipCopy = true;
414  if (!skipCopy) {
415  // The copy happens right before the op that is bufferized.
416  builder.setInsertionPoint(op);
417  createMemCpy(builder, loc, operandBuffer, resultBuffer);
418  }
419  return resultBuffer;
420  }
421 
422  // Bufferizing in-place. No need to allocate a new buffer.
423  return operandBuffer;
424 }

426 void mlir::linalg::comprehensive_bufferize::BufferizationState::replaceOp(
427  Operation *op, ValueRange values) {
428  OpBuilder &b = getBuilder();
429  OpBuilder::InsertionGuard g(b);
430 
431  // Replace all OpResults with the given values.
432  for (OpResult opResult : op->getOpResults()) {
433  // Skip OpResult if it has no uses.
434  if (opResult.getUses().empty())
435  continue;
436 
437  Value replacement = values[opResult.getResultNumber()];
438  if (opResult.getType().isa<TensorType>()) {
439  // The OpResult is a tensor. Such values are replaced with memrefs during
440  // bufferization.
441  assert((replacement.getType().isa<MemRefType>() ||
442  replacement.getType().isa<UnrankedMemRefType>()) &&
443  "tensor op result should be replaced with a memref value");
444  // The existing uses of the OpResult still expect a tensor. Insert a
445  // ToTensorOp. Throughout bufferization, this ToTensorOp will gradually
446  // loose all of its users and eventually DCE away.
447  setInsertionPointAfter(b, replacement);
448  replacement = b.create<bufferization::ToTensorOp>(replacement.getLoc(),
449  replacement);
450  }
451  opResult.replaceAllUsesWith(replacement);
452  }
453 
454  op->erase();
455 }

466 LogicalResult
467 mlir::linalg::comprehensive_bufferize::bufferize(Block *block,
468  BufferizationState &state) {
469  // Ops may get deleted during the traversal, so do not iterate over `block`
470  // directly.
471  SmallVector<Operation *> ops;
472  ops.reserve(block->getOperations().size());
473  for (Operation &op : *block)
474  ops.push_back(&op);
475  for (Operation *op : ops)
476  if (failed(bufferize(op, state)))
477  return failure();
478  return success();
479 }

481 LogicalResult
482 mlir::linalg::comprehensive_bufferize::bufferize(Operation *op,
483  BufferizationState &state) {
484  OpBuilder &b = state.getBuilder();
485 
486  // Check if op has tensor results or operands.
487  auto isaTensor = [](Type t) { return t.isa<TensorType>(); };
488  bool hasTensorResult = any_of(op->getResultTypes(), isaTensor);
489  bool hasTensorOperand = any_of(op->getOperandTypes(), isaTensor);
490  bool hasRegions = !op->getRegions().empty();
491 
492  // No tensor results/operands or regions. We are done.
493  if (!hasTensorResult && !hasTensorOperand && !hasRegions)
494  return success();
495 
496  // Bufferize using `BufferizableOpInterface`. Interface implementations are
497  // responsible for bufferizing nested ops.
498  if (auto bufferizableOp = state.getOptions().dynCastBufferizableOp(op)) {
499  b.setInsertionPoint(op);
500  return bufferizableOp.bufferize(b, state);
501  }
502 
503  // `op` is an unbufferizable tensor op.
504  if (!state.getOptions().allowUnknownOps)
505  return op->emitError() << "unsupported op with tensors";
506 
507  // Bufferize all regions.
508  for (Region &region : op->getRegions())
509  if (failed(bufferize(&region, state)))
510  return failure();
511 
512  return success();
513 }

519 /// Move the insertion point of the given builder to the beginning of a
520 /// surrounding block as much as possible, while not crossing any allocation
521 /// hoisting barriers.
522 static void moveInsertionPointToAllocationHoistingBarrier(OpBuilder &b) {
523  Operation *op = b.getInsertionBlock()->getParentOp();
524  while (op) {
525  if (auto bufferizableOp = dyn_cast<BufferizableOpInterface>(op))
526  if (bufferizableOp.isAllocationHoistingBarrier())
527  break;
528  op = op->getParentOp();
529  }
530 
531  if (!op) {
532  // No allocation hoisting barrier found. Hoist to FuncOp.
533  op = b.getInsertionBlock()->getParentOp();
534  if (!isa<FuncOp>(op))
535  op = op->getParentOfType<FuncOp>();
536  assert(op && "could not find enclosing FuncOp");
537  }
538 
539  // TODO: Handle cases where allocation hoisting barrier has more than one
540  // region or block.
541  assert(op->getNumRegions() == 1 &&
542  "allocation hoisting barriers with >1 regions not supported");
543  assert(op->getRegion(0).getBlocks().size() == 1 &&
544  "allocation hoisting barriers with >1 blocks not supported");
545  b.setInsertionPointToStart(&(op->getRegion(0).front()));
546 }

548 /// Compute the type of the `memref` to use for allocating the buffer for
549 /// `shapedValue`. Also returns (by reference in `dynShape`), the value for the
550 /// dynamic dimensions in the returned `memref` type. The function may also set
551 /// the insertion point to an earlier location, where the allocation should
552 /// happen ("allocation hoisting").
553 static MemRefType getAllocationTypeAndShape(OpBuilder &b, Location loc,
554  Value shapedValue,
555  SmallVectorImpl<Value> &dynShape) {
556  MemRefType allocMemRefType =
557  getContiguousMemRefType(shapedValue.getType().cast<ShapedType>());
558 
559  // Compute the dynamic part of the shape.
560  bool reifiedShapes = false;
561  if (auto rankedOp = dyn_cast_or_null<ReifyRankedShapedTypeOpInterface>(
562  shapedValue.getDefiningOp())) {
563  ReifiedRankedShapedTypeDims resultDims;
564  if (succeeded(rankedOp.reifyResultShapes(b, resultDims))) {
565  reifiedShapes = true;
566  OpResult resultValue = shapedValue.dyn_cast<OpResult>();
567  auto &shape = resultDims[resultValue.getResultNumber()];
568  for (const auto &dim : enumerate(allocMemRefType.getShape()))
569  if (ShapedType::isDynamic(dim.value()))
570  dynShape.push_back(shape[dim.index()]);
571  }
572  }
573 
574  if (!reifiedShapes) {
575  for (const auto &dim : enumerate(allocMemRefType.getShape()))
576  if (ShapedType::isDynamic(dim.value())) {
577  assert((shapedValue.getType().isa<UnrankedMemRefType>() ||
578  shapedValue.getType().isa<MemRefType>()) &&
579  "expected MemRef type");
580  dynShape.push_back(
581  b.create<memref::DimOp>(loc, shapedValue, dim.index()));
582  }
583  }
584 
585  // If the buffer is statically shaped, try to hoist it to the first enclosing
586  // parallel region.
587  // TODO: also hoist in the dynamic case. For now this relies on subsequent
588  // calls to LICM and buffer hoisting which will most likely not succeed.
589  // TODO: when packing, allocate a static bounding box which will enable more
590  // hoisting.
591  if (dynShape.empty())
592  moveInsertionPointToAllocationHoistingBarrier(b);
593 
594  return allocMemRefType;
595 }

597 /// Create an Allocop/DeAllocOp pair, where the AllocOp is after
598 /// `shapedValue.getDefiningOp` (or at the top of the block in case of a
599 /// bbArg) and the DeallocOp is at the end of the block.
600 Value mlir::linalg::comprehensive_bufferize::BufferizationState::
601  createAllocDeallocPair(OpBuilder &b, Location loc, Value shapedValue) {
602  // Take a guard before anything else.
