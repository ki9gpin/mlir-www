84  castOp.getResult().getType(), layout, memorySpace);
85  state.replaceOpWithNewOp<memref::CastOp>(b, op, memRefType, resultBuffer);

145  LogicalResult bufferize(Operation *op, OpBuilder &b,
146  BufferizationState &state) const {
147  auto extractSliceOp = cast<tensor::ExtractSliceOp>(op);
148  Location loc = extractSliceOp.getLoc();
149  Value srcMemref = state.lookupBuffer(extractSliceOp.source());
150  auto srcMemrefType = srcMemref.getType().cast<MemRefType>();
151  auto dstTensorType =
152  extractSliceOp.result().getType().cast<RankedTensorType>();
153 
154  // If not inplaceable, alloc.
155  bool inplace = state.isInPlace(extractSliceOp->getResult(0));
156  Value alloc;
157  if (!inplace)
158  alloc = state.createAllocDeallocPair(b, loc, extractSliceOp.result());
159 
160  // Bufferize to subview.
161  auto subviewMemRefType =
162  memref::SubViewOp::inferRankReducedResultType(
163  dstTensorType.getRank(), srcMemrefType,
164  extractSliceOp.getMixedOffsets(), extractSliceOp.getMixedSizes(),
165  extractSliceOp.getMixedStrides())
166  .cast<MemRefType>();
167  Value subView = b.create<memref::SubViewOp>(
168  loc, subviewMemRefType, srcMemref, extractSliceOp.getMixedOffsets(),
169  extractSliceOp.getMixedSizes(), extractSliceOp.getMixedStrides());
170 
171  /// If not inplaceable, copy.
172  if (!inplace) {
173  // Do not copy if the copied data is never read.
174  if (state.isValueRead(extractSliceOp.result()))
175  state.createMemCpy(b, extractSliceOp.getLoc(), subView, alloc);
176  subView = alloc;
177  }
178 
179  state.replaceOp(op, subView);
180  return success();
181  }

256 /// Return true if the (ExtractSliceOp, InsertSliceOp) pair match (i.e.
257 /// equivalent operand / result and same offset/sizes/strides specification).
258 ///
259 /// This is one particular type of relationship between ops on tensors that
260 /// reduce to an equivalence on buffers. This should be generalized and
261 /// exposed as interfaces on the proper types.
262 static bool
263 areEquivalentExtractSliceOps(const BufferizationAliasInfo &aliasInfo,
264  ExtractSliceOp st, InsertSliceOp sti) {
265  if (!st || !sti)
266  return false;
267  if (!aliasInfo.areEquivalentBufferizedValues(st.source(), sti.dest()))
268  return false;
269  if (!sameOffsetsSizesAndStrides(st, sti, isEqualConstantIntOrValue))
270  return false;
271  return true;
272 }

274 /// Return true if the source of a `insertSliceOp` bufferizes to an
275 /// equivalent ExtractSliceOp that bufferizes inplace.
276 static bool isSourceEquivalentToAMatchingInplaceExtractSliceOp(
277  const BufferizationAliasInfo &aliasInfo, InsertSliceOp insertSliceOp) {
278  bool foundOp = false;
279  aliasInfo.applyOnEquivalenceClass(insertSliceOp.source(), [&](Value value) {
280  auto extractSliceOp = value.getDefiningOp<ExtractSliceOp>();
281  if (extractSliceOp &&
282  areEquivalentExtractSliceOps(aliasInfo, extractSliceOp,
283  insertSliceOp) &&
284  aliasInfo.isInPlace(extractSliceOp->getResult(0))) {
285  foundOp = true;
286  }
287  });
288  return foundOp;
289 }

291 /// Return true if `value` is originating from an ExtractSliceOp that matches
292 /// the given InsertSliceOp.
293 static bool hasMatchingExtractSliceOp(const BufferizationAliasInfo &aliasInfo,
294  BufferizationState &state, Value value,
295  InsertSliceOp insertOp) {
296  auto condition = [&](Value val) {
297  if (auto extractOp = val.getDefiningOp<ExtractSliceOp>())
298  if (areEquivalentExtractSliceOps(aliasInfo, extractOp, insertOp))
299  return true;
300  return false;
301  };
302 
303  return llvm::all_of(state.findValueInReverseUseDefChain(value, condition),
304  condition);
305 }

333  bool isNotConflicting(Operation *op, OpOperand *uRead,
334  OpOperand *uConflictingWrite, BufferizationState &state,
335  const BufferizationAliasInfo &aliasInfo) const {
336  Operation *readingOp = uRead->getOwner();
337  Operation *conflictingWritingOp = uConflictingWrite->getOwner();
338 
339  // Special rules for matching ExtractSliceOp/InsertSliceOp pairs. If
340  // uRead is an InsertSliceOp...
341  if (auto insertSliceOp = dyn_cast<InsertSliceOp>(readingOp)) {
342  // As an example, consider the following IR.
343  //
344  // %0 = tensor.extract_slice %t[%a, %b][%c, %d][1, 1] {inplace = [true] }
345  // %1 = linalg.fill %cst, %0 {inplace= [true] }
346  // %2 = tensor.insert_slice %1 into %t[%a, %b][%c, %d][1, 1]
347  // {inplace= [true] }
348 
349  // TODO: Use insertSliceOp.getDestOpOperand etc. when available.
350  if (uRead == &insertSliceOp->getOpOperand(1) /*dest*/ &&
351  hasMatchingExtractSliceOp(aliasInfo, state, uConflictingWrite->get(),
352  insertSliceOp))
353  // Case 1: The main insight is that InsertSliceOp reads only part of
354  // the destination tensor. The overwritten area is not read. If
355  // uConflictingWrite writes into exactly the memory location that is
356  // being read by uRead, this is not a conflict.
357  //
358  // In the above example:
359  // uRead = OpOperand 1 (%t) of tensor.insert_slice
360  // uConflictingWrite = OpOperand 1 (%0) of linalg.fill
361  //
362  // The read of %t does not conflict with the write of the FillOp
363  // (same aliases!) because the area that the FillOp operates on is
364  // exactly the one that is *not* read via %t.
365  return true;
366 
367  if (uRead == &insertSliceOp->getOpOperand(0) /*source*/ &&
368  uConflictingWrite == &insertSliceOp->getOpOperand(1) /*dest*/ &&
369  hasMatchingExtractSliceOp(aliasInfo, state, uRead->get(),
370  insertSliceOp))
371  // Case 2: The read of the source tensor and the write to the dest
372  // tensor via an InsertSliceOp is not a conflict if the read is
373  // reading exactly that part of an equivalent tensor that the
374  // InsertSliceOp is writing.
375  //
376  // In the above example:
377  // uRead = OpOperand 0 (%1) of tensor.insert_slice
378  // uConflictingWrite = OpOperand 1 (%t) of tensor.insert_slice
379  return true;
380  }
381 
382  // If uConflictingWrite is an InsertSliceOp...
383  if (auto insertSliceOp = dyn_cast<InsertSliceOp>(conflictingWritingOp))
384  // As an example, consider the following IR.
385  //
386  // %0 = tensor.extract_slice %t[%a, %b][%c, %d][1, 1] {inplace = [true] }
387  // %1 = linalg.fill %cst, %0 {inplace= [true] }
388  // %2 = tensor.insert_slice %1 into %t[%a, %b][%c, %d][1, 1]
389  // {inplace= [true] }
390  // %3 = vector.transfer_read %1, %cst
391  //
392  // In the above example:
393  // uRead = OpOperand 0 (%1) of vector.transfer_read
394  // uConflictingWrite = OpOperand 1 (%t) of tensor.insert_slice
395  // lastWrite = %1
396  //
397  // This is not a conflict because the InsertSliceOp overwrites the
398  // memory segment of %1 with the exact same data. (Effectively, there
399  // is no memory write here.)
400  if (uConflictingWrite == &insertSliceOp->getOpOperand(1) /*dest*/ &&
401  aliasInfo.areEquivalentBufferizedValues(uRead->get(),
402  insertSliceOp.source()) &&
403  hasMatchingExtractSliceOp(aliasInfo, state, insertSliceOp.source(),
404  insertSliceOp))
405  return true;
406 
407  return false;
408  }

410  LogicalResult bufferize(Operation *op, OpBuilder &b,
411  BufferizationState &state) const {
412  // insert_slice ops arise from tiling and bufferizing them out-of-place is
413  // generally a deal breaker. When used with loops, this ends up cloning the
414  // whole tensor on every single iteration and is a symptom of a
415  // catastrophically bad scheduling decision.
416  // TODO: be very loud about it or even consider failing the pass.
417  auto insertSliceOp = cast<tensor::InsertSliceOp>(op);
418  Location loc = insertSliceOp.getLoc();
419  TensorBufferizationState &tensorState = getTensorBufferizationState(state);
420 
421  // When bufferizing out-of-place, `getResultBuffer` allocates.
422  Value dstMemref = state.getResultBuffer(insertSliceOp->getResult(0));
423  if (!dstMemref)
424  return failure();
425 
426  bool needCopy =
427  !tensorState.insertSliceOpsWithoutCopy.contains(insertSliceOp);
428  if (needCopy) {
429  // Take a subview of the dst.
430  auto dstMemrefType = dstMemref.getType().cast<MemRefType>();
431  auto subviewMemRefType =
432  memref::SubViewOp::inferRankReducedResultType(
433  insertSliceOp.getSourceType().getRank(), dstMemrefType,
434  insertSliceOp.getMixedOffsets(), insertSliceOp.getMixedSizes(),
435  insertSliceOp.getMixedStrides())
436  .cast<MemRefType>();
437  Value subView = b.create<memref::SubViewOp>(
438  loc, subviewMemRefType, dstMemref, insertSliceOp.getMixedOffsets(),
439  insertSliceOp.getMixedSizes(), insertSliceOp.getMixedStrides());
440  // Copy tensor.
441  Value srcMemref = state.lookupBuffer(insertSliceOp.source());
442  state.createMemCpy(b, insertSliceOp.getLoc(), srcMemref, subView);
443  }
444 
445  state.replaceOp(op, dstMemref);
446  return success();
447  }

455 LogicalResult mlir::linalg::comprehensive_bufferize::tensor_ext::
456  InplaceInsertSliceOpAnalysis::run(Operation *op, BufferizationState &state,
457  BufferizationAliasInfo &aliasInfo,
458  SmallVector<Operation *> &newOps) {
459  auto &tensorState = getTensorBufferizationState(state);
460  op->walk([&](InsertSliceOp insertSliceOp) {
461  // A copy of the source buffer is needed if either:
462  // - The producer of `source` is not inplace. This is the case where a
463  // slice is computed out of place into the inplace full tensor.
464  // - The result is not inplace. This is the case where the whole tensor is
465  // cloned and the clone needs to be updated.
466  if (isSourceEquivalentToAMatchingInplaceExtractSliceOp(aliasInfo,
467  insertSliceOp) &&
468  state.isInPlace(insertSliceOp->getResult(0)))
469  tensorState.insertSliceOpsWithoutCopy.insert(insertSliceOp);
470  });
471  return success();
472 }
