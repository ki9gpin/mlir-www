38 void IntegerPolyhedron::append(const IntegerPolyhedron &other) {
39  assert(other.getNumCols() == getNumCols());
40  assert(other.getNumDimIds() == getNumDimIds());
41  assert(other.getNumSymbolIds() == getNumSymbolIds());
42 
43  inequalities.reserveRows(inequalities.getNumRows() +
44  other.getNumInequalities());
45  equalities.reserveRows(equalities.getNumRows() + other.getNumEqualities());
46 
47  for (unsigned r = 0, e = other.getNumInequalities(); r < e; r++) {
48  addInequality(other.getInequality(r));
49  }
50  for (unsigned r = 0, e = other.getNumEqualities(); r < e; r++) {
51  addEquality(other.getEquality(r));
52  }
53 }

128 void IntegerPolyhedron::removeIdRange(unsigned idStart, unsigned idLimit) {
129  assert(idLimit < getNumCols() && "invalid id limit");
130 
131  if (idStart >= idLimit)
132  return;
133 
134  // We are going to be removing one or more identifiers from the range.
135  assert(idStart < numIds && "invalid idStart position");
136 
137  // TODO: Make 'removeIdRange' a lambda called from here.
138  // Remove eliminated identifiers from the constraints..
139  equalities.removeColumns(idStart, idLimit - idStart);
140  inequalities.removeColumns(idStart, idLimit - idStart);
141 
142  // Update members numDims, numSymbols and numIds.
143  unsigned numDimsEliminated = 0;
144  unsigned numLocalsEliminated = 0;
145  unsigned numColsEliminated = idLimit - idStart;
146  if (idStart < numDims) {
147  numDimsEliminated = std::min(numDims, idLimit) - idStart;
148  }
149  // Check how many local id's were removed. Note that our identifier order is
150  // [dims, symbols, locals]. Local id start at position numDims + numSymbols.
151  if (idLimit > numDims + numSymbols) {
152  numLocalsEliminated = std::min(
153  idLimit - std::max(idStart, numDims + numSymbols), getNumLocalIds());
154  }
155  unsigned numSymbolsEliminated =
156  numColsEliminated - numDimsEliminated - numLocalsEliminated;
157 
158  numDims -= numDimsEliminated;
159  numSymbols -= numSymbolsEliminated;
160  numIds = numIds - numColsEliminated;
161 }

183 void IntegerPolyhedron::swapId(unsigned posA, unsigned posB) {
184  assert(posA < getNumIds() && "invalid position A");
185  assert(posB < getNumIds() && "invalid position B");
186 
187  if (posA == posB)
188  return;
189 
190  for (unsigned r = 0, e = getNumInequalities(); r < e; r++)
191  std::swap(atIneq(r, posA), atIneq(r, posB));
192  for (unsigned r = 0, e = getNumEqualities(); r < e; r++)
193  std::swap(atEq(r, posA), atEq(r, posB));
194 }

221 /// Gather all lower and upper bounds of the identifier at `pos`, and
222 /// optionally any equalities on it. In addition, the bounds are to be
223 /// independent of identifiers in position range [`offset`, `offset` + `num`).
224 void IntegerPolyhedron::getLowerAndUpperBoundIndices(
225  unsigned pos, SmallVectorImpl<unsigned> *lbIndices,
226  SmallVectorImpl<unsigned> *ubIndices, SmallVectorImpl<unsigned> *eqIndices,
227  unsigned offset, unsigned num) const {
228  assert(pos < getNumIds() && "invalid position");
229  assert(offset + num < getNumCols() && "invalid range");
230 
231  // Checks for a constraint that has a non-zero coeff for the identifiers in
232  // the position range [offset, offset + num) while ignoring `pos`.
233  auto containsConstraintDependentOnRange = [&](unsigned r, bool isEq) {
234  unsigned c, f;
235  auto cst = isEq ? getEquality(r) : getInequality(r);
236  for (c = offset, f = offset + num; c < f; ++c) {
237  if (c == pos)
238  continue;
239  if (cst[c] != 0)
240  break;
241  }
242  return c < f;
243  };
244 
245  // Gather all lower bounds and upper bounds of the variable. Since the
246  // canonical form c_1*x_1 + c_2*x_2 + ... + c_0 >= 0, a constraint is a lower
247  // bound for x_i if c_i >= 1, and an upper bound if c_i <= -1.
248  for (unsigned r = 0, e = getNumInequalities(); r < e; r++) {
249  // The bounds are to be independent of [offset, offset + num) columns.
250  if (containsConstraintDependentOnRange(r, /*isEq=*/false))
251  continue;
252  if (atIneq(r, pos) >= 1) {
253  // Lower bound.
254  lbIndices->push_back(r);
255  } else if (atIneq(r, pos) <= -1) {
256  // Upper bound.
257  ubIndices->push_back(r);
258  }
259  }
260 
261  // An equality is both a lower and upper bound. Record any equalities
262  // involving the pos^th identifier.
263  if (!eqIndices)
264  return;
265 
266  for (unsigned r = 0, e = getNumEqualities(); r < e; r++) {
267  if (atEq(r, pos) == 0)
268  continue;
269  if (containsConstraintDependentOnRange(r, /*isEq=*/true))
270  continue;
271  eqIndices->push_back(r);
272  }
273 }

275 bool IntegerPolyhedron::hasConsistentState() const {
276  if (!inequalities.hasConsistentState())
277  return false;
278  if (!equalities.hasConsistentState())
279  return false;
280 
281  // Catches errors where numDims, numSymbols, numIds aren't consistent.
282  if (numDims > numIds || numSymbols > numIds || numDims + numSymbols > numIds)
283  return false;
284 
285  return true;
286 }
