89  LogicalResult bufferize(Operation *op, OpBuilder &b,
90  BufferizationState &state) const {
91  auto writeOp = cast<vector::TransferWriteOp>(op);
92  assert(writeOp.getShapedType().isa<TensorType>() &&
93  "only tensor types expected");
94 
95  // Create a new transfer_write on buffer that doesn't have a return value.
96  // Leave the previous transfer_write to dead code as it still has uses at
97  // this point.
98  Value resultBuffer = state.getResultBuffer(op->getResult(0));
99  if (!resultBuffer)
100  return failure();
101  b.create<vector::TransferWriteOp>(
102  writeOp.getLoc(), writeOp.vector(), resultBuffer, writeOp.indices(),
103  writeOp.permutation_mapAttr(), writeOp.in_boundsAttr());
104  state.replaceOp(op, resultBuffer);
